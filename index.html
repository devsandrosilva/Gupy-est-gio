import json # 1) Cálculo da SOMA def calcular_soma(indice): soma = 0 for k in range(1, indice + 1): soma += k return soma # 2) Verificação na sequência de Fibonacci def pertence_fibonacci(n): a, b = 0, 1 while a < n: a, b = b, a + b return a == n # 3) Faturamento diário def faturamento_diario(dados): dias_faturamento = [valor for valor in dados if valor > 0] media = sum(dias_faturamento) / len(dias_faturamento) menor = min(dias_faturamento) maior = max(dias_faturamento) dias_acima_media = len([dia for dia in dias_faturamento if dia > media]) return menor, maior, dias_acima_media # 4) Percentual de faturamento por estado def percentual_por_estado(faturamento): total = sum(faturamento.values()) percentuais = {estado: (valor / total) * 100 for estado, valor in faturamento.items()} return percentuais # 5) Inversão de uma string def inverter_string(s): inversa = '' for char in s: inversa = char + inversa return inversa # Exemplo de uso if __name__ == "__main__": # 1) Cálculo da SOMA indice = 13 soma = calcular_soma(indice) print(f"SOMA de 1 até {indice}: {soma}") #
